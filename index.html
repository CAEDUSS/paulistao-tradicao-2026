<script>

const equipes = [

  "Legião Sorocabana",

  "Romanos",

  "Milites Peregrinus",

  "Pão de Queijo",

  "Galícia",

  "Oratório",

  "Trad Fut"

];

let tabela = {};

let jogos = [];

function init() {

  equipes.forEach(e => {

    tabela[e] = { pts:0,j:0,v:0,e:0,d:0,sg:0 };

  });

  gerarJogos();

  renderRodadas();

  renderTabela();

}

function gerarJogos() {

  for (let i=0;i<equipes.length;i++){

    for (let j=i+1;j<equipes.length;j++){

      jogos.push({t1:equipes[i], t2:equipes[j], g1:null, g2:null});

    }

  }

}

function renderRodadas() {

  const div = document.getElementById("rodadas");

  div.innerHTML = "";

  jogos.forEach((jogo, idx) => {

    div.innerHTML += `

      <div class="rodada">

        ${jogo.t1}

        <input type="number" id="g1-${idx}"> x

        <input type="number" id="g2-${idx}">

        ${jogo.t2}

        <button onclick="salvar(${idx})">OK</button>

      </div>

    `;

  });

}

function salvar(i) {

  const g1 = parseInt(document.getElementById(`g1-${i}`).value);

  const g2 = parseInt(document.getElementById(`g2-${i}`).value);

  if (isNaN(g1) || isNaN(g2)) return;

  const j = jogos[i];

  j.g1 = g1;

  j.g2 = g2;

  atualizarTabela();

}

function atualizarTabela() {

  equipes.forEach(e => tabela[e] = { pts:0,j:0,v:0,e:0,d:0,sg:0 });

  jogos.forEach(j => {

    if (j.g1 === null) return;

    let t1 = tabela[j.t1];

    let t2 = tabela[j.t2];

    t1.j++; t2.j++;

    t1.sg += j.g1 - j.g2;

    t2.sg += j.g2 - j.g1;

    if (j.g1 > j.g2) { t1.v++; t1.pts+=3; t2.d++; }

    else if (j.g2 > j.g1) { t2.v++; t2.pts+=3; t1.d++; }

    else { t1.e++; t2.e++; t1.pts++; t2.pts++; }

  });

  renderTabela();

  gerarMataMata();

}

function renderTabela() {

  const tbody = document.querySelector("#tabela tbody");

  tbody.innerHTML = "";

  let ranking = Object.entries(tabela).sort((a,b)=>b[1].pts - a[1].pts || b[1].sg - a[1].sg);

  ranking.forEach((r,i)=>{

    let d = r[1];

    tbody.innerHTML += `

      <tr>

        <td>${i+1}</td>

        <td>${r[0]}</td>

        <td>${d.pts}</td>

        <td>${d.j}</td>

        <td>${d.v}</td>

        <td>${d.e}</td>

        <td>${d.d}</td>

        <td>${d.sg}</td>

      </tr>

    `;

  });

}

function gerarMataMata() {

  const div = document.getElementById("mata");

  let ranking = Object.entries(tabela).sort((a,b)=>b[1].pts - a[1].pts || b[1].sg - a[1].sg);

  if (ranking[3][1].pts === 0) {

    div.innerHTML = "<p>Preencha os jogos para gerar semifinais</p>";

    return;

  }

  let semi1 = `${ranking[0][0]} x ${ranking[3][0]}`;

  let semi2 = `${ranking[1][0]} x ${ranking[2][0]}`;

  div.innerHTML = `

    <div class="rodada">

      <strong>Semifinais</strong><br>

      ${semi1}<br>

      ${semi2}

    </div>

  `;

}

init();
